**🚀 Quick Start:**
```bash
npm install @synerity/ui
```

```typescript
import { SuiButton, SuiCard } from '@synerity/ui';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [SuiButton, SuiCard],
  template: `
    <sui-card>
      <sui-button variant="primary">Get Started</sui-button>
    </sui-card>
  `
})
export class ExampleComponent {}
```
