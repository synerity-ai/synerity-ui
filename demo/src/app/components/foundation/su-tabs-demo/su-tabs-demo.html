API Reference

Properties

@Input() activeIndex: number = 0; // Index of the currently active tab @Input() orientation: 'top' | 'bottom' | 'left' | 'right' = 'top'; // Tab orientation @Input() scrollable: boolean = false; // Enable scrolling for many tabs @Input() style: any = {}; // Custom styles @Input() styleClass: string = ''; // Custom CSS classes @Input() tabs: TabItem[] = []; // Array of tab items @Output() onActiveIndexChange: EventEmitter<number>; // Active tab change event @Output() onTabClose: EventEmitter<number>; // Tab close event

TabItem Interface

interface TabItem { header: string; // Tab header text content?: any; // Tab content (optional) disabled?: boolean; // Disable the tab closable?: boolean; // Show close button icon?: string; // Tab icon }

Methods

setActiveIndex(index: number): void; // Set active tab by index closeTab(index: number): void; // Close tab by index isTabActive(index: number): boolean; // Check if tab is active isTabDisabled(index: number): boolean; // Check if tab is disabled getTabClass(index: number): string; // Get tab CSS classes getTabsClass(): string; // Get tabs container CSS classes getTabsStyle(): any; // Get tabs container styles

CSS Classes

.sui-tabs // Base tabs class .sui-tabs-top // Top orientation .sui-tabs-bottom // Bottom orientation .sui-tabs-left // Left orientation .sui-tabs-right // Right orientation .sui-tabs-tab // Individual tab class .sui-tabs-tab-active // Active tab class .sui-tabs-tab-disabled // Disabled tab class
