API Reference

Properties

@Input() value: string | number = ''; // Badge value @Input() severity: 'info' | 'success' | 'warning' | 'danger' | 'contrast' = 'info'; @Input() size: 'small' | 'normal' | 'large' = 'normal'; @Input() style: any = {{ '{' }}{{ '}' }}; // Custom styles @Input() styleClass: string = ''; // Custom CSS classes

Methods

getBadgeClass(): string; // Get computed CSS classes getBadgeStyle(): any; // Get computed styles hasValue(): boolean; // Check if badge has value getDisplayValue(): string; // Get display value (99+ for large numbers)
