API Reference

Properties

@Input() tabs: AccordionTab[]; // Array of accordion tabs @Input() activeIndex: number | number[]; // Active tab index(es) @Input() multiple: boolean = false; // Allow multiple open tabs @Input() variant: 'default' | 'bordered' | 'minimal' = 'default'; @Input() size: 'small' | 'medium' | 'large' = 'medium'; @Input() style: any = {{ '{' }}{{ '}' }}; // Custom styles @Input() styleClass: string = ''; // Custom CSS classes

Events

@Output() onActiveIndexChange: EventEmitter<number | number[]>; @Output() onOpen: EventEmitter<AccordionTab>; @Output() onClose: EventEmitter<AccordionTab>;
