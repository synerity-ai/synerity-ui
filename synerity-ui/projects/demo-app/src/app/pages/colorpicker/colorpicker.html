ColorPicker

Color selection components with visual preview, format support, and form integration.

Basic Usage

<!-- Basic Color -->
<sui-color-picker 
  [(ngModel)]="basicColor"
  (onChange)="onColorChange($event, 'basic')">
</sui-color-picker>

<!-- Primary Color -->
<sui-color-picker 
  [(ngModel)]="primaryColor"
  (onChange)="onColorChange($event, 'primary')">
</sui-color-picker>

<!-- Secondary Color -->
<sui-color-picker 
  [(ngModel)]="secondaryColor"
  (onChange)="onColorChange($event, 'secondary')">
</sui-color-picker>

Color States

<!-- Normal State -->
<sui-color-picker 
  [(ngModel)]="basicColor">
</sui-color-picker>

<!-- Disabled State -->
<sui-color-picker 
  [(ngModel)]="disabledColor"
  [disabled]="true">
</sui-color-picker>

<!-- With Value -->
<sui-color-picker 
  [(ngModel)]="customColor">
</sui-color-picker>
