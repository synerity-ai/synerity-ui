Sidebar

A navigation sidebar component with menu items and submenus.

Basic Usage

(click)="toggleBasicSidebar()"
<sui-sidebar 
  [visible]="showBasicSidebar"
  [items]="sidebarItems"
  (onShow)="onShow()"
  (onHide)="onHide()"
  (visibleChange)="showBasicSidebar = $event"
>
</sui-sidebar>

Positioned Sidebar

<sui-sidebar 
  [visible]="showPositionedSidebar"
  [position]="'right'"
  [items]="sidebarItems"
  (visibleChange)="showPositionedSidebar = $event"
>
</sui-sidebar>

Modal Sidebar

<sui-sidebar 
  [visible]="showModalSidebar"
  [modal]="true"
  [fullScreen]="true"
  [items]="sidebarItems"
  (visibleChange)="showModalSidebar = $event"
>
</sui-sidebar>

Custom Styled Sidebar

<sui-sidebar 
  [visible]="showCustomSidebar"
  [style]="{
    backgroundColor: '#f0f9ff',
    borderColor: '#0ea5e9'
  }"
  [styleClass]="'sui-sidebar-small'"
  [items]="sidebarItems"
  (visibleChange)="showCustomSidebar = $event"
>
</sui-sidebar>
