File Upload Advanced

An advanced file upload component with drag & drop, progress tracking, and file management.

Basic File Upload

<sui-file-upload-advanced 
  [multiple]="true"
  [accept]="'image/*,.pdf,.doc,.docx'"
  [maxFileSize]="5 * 1024 * 1024"
  [maxFiles]="5"
  [autoUpload]="false"
  (onSelect)="onFileSelect($event)"
  (onUpload)="onFileUpload($event)"
  (onRemove)="onFileRemove($event)"
  (onError)="onFileError($event)"
></sui-file-upload-advanced>

Drag & Drop with Auto Upload

<sui-file-upload-advanced 
  [multiple]="true"
  [accept]="'image/*,video/*'"
  [maxFileSize]="10 * 1024 * 1024"
  [maxFiles]="3"
  [dragDropEnabled]="true"
  [autoUpload]="true"
  [showProgress]="true"
  (onSelect)="onFileSelect($event)"
  (onUpload)="onFileUpload($event)"
  (onRemove)="onFileRemove($event)"
  (onError)="onFileError($event)"
></sui-file-upload-advanced>

Custom Styled Upload

<sui-file-upload-advanced 
  [multiple]="false"
  [accept]="'.pdf'"
  [maxFileSize]="2 * 1024 * 1024"
  [maxFiles]="1"
  [dragDropEnabled]="true"
  [showPreview]="false"
  [style]="{
    borderColor: '#10b981',
    backgroundColor: '#f0fdf4'
  }"
  [styleClass]="'sui-file-upload-advanced-large'"
  (onSelect)="onFileSelect($event)"
  (onUpload)="onFileUpload($event)"
  (onRemove)="onFileRemove($event)"
  (onError)="onFileError($event)"
></sui-file-upload-advanced>
