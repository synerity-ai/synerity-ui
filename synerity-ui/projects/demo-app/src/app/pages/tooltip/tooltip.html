Tooltip

Hover-triggered help text that provides additional information about UI elements with customizable positioning and timing.

Basic Tooltip

<sui-tooltip
  [tooltip]="basicTooltipConfig.tooltip"
  [position]="basicTooltipConfig.position"
  [showDelay]="basicTooltipConfig.showDelay"
  [hideDelay]="basicTooltipConfig.hideDelay"
  [disabled]="basicTooltipConfig.disabled"
  (onShow)="onTooltipShow()"
  (onHide)="onTooltipHide()">
  
  <button>Hover me</button>
</sui-tooltip>

Positioned Tooltips

<sui-tooltip
  [tooltip]="topTooltipConfig.tooltip"
  [position]="'top'"
  [showDelay]="0"
  [hideDelay]="0"
  [disabled]="false">
  
  <button>Top</button>
</sui-tooltip>
