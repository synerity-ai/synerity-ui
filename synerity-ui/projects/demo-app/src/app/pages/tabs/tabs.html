<sui-tabs 
  [tabs]="basicTabs"
  [activeIndex]="basicActiveIndex"
  (onActiveIndexChange)="onBasicActiveIndexChange($event)">
  
  <div *ngIf="basicActiveIndex === 0" class="space-y-4">
    <h4 class="text-lg font-medium text-gray-900">First Tab Content</h4>
    <p class="text-gray-600">This is the content of the first tab.</p>
  </div>
  
  <div *ngIf="basicActiveIndex === 1" class="space-y-4">
    <h4 class="text-lg font-medium text-gray-900">Second Tab Content</h4>
    <p class="text-gray-600">This is the content of the second tab.</p>
  </div>
</sui-tabs>

<sui-tabs 
  [tabs]="closableTabs"
  [activeIndex]="closableActiveIndex"
  (onActiveIndexChange)="onClosableActiveIndexChange($event)"
  (onTabClose)="onTabClose($event)">
  
  <div *ngIf="closableActiveIndex >= 0 && closableTabs[closableActiveIndex]" class="space-y-4">
    <h4 class="text-lg font-medium text-gray-900">{{ closableTabs[closableActiveIndex].header }} Content</h4>

  </div>
</sui-tabs>

Installation

The Tabs component is part of the Synerity UI library. Install it using npm:

npm install @synerity/ui

Props

Prop Type Default Description
activeIndex number 0 Index of the currently active tab
orientation 'top' | 'bottom' | 'left' | 'right' 'top' Position of the tab navigation
scrollable boolean false Whether tabs should be scrollable when they overflow
