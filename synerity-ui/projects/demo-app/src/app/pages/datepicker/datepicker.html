DatePicker

Date selection components with calendar popup, validation, and form integration.

Basic Usage

<!-- Basic Date -->
<sui-date-picker 
  [(ngModel)]="basicDate"
  placeholder="Select date"
  name="basicDate"
  (onChange)="onDateChange($event, 'basic')">
</sui-date-picker>

<!-- With Placeholder -->
<sui-date-picker 
  [(ngModel)]="placeholderDate"
  placeholder="Choose a date..."
  name="placeholderDate">
</sui-date-picker>

<!-- With Icon -->
<sui-date-picker 
  [(ngModel)]="basicDate"
  [showIcon]="true"
  placeholder="Date with icon"
  name="iconDate">
</sui-date-picker>

Date States

<!-- Normal State -->
<sui-date-picker 
  [(ngModel)]="basicDate"
  placeholder="Normal date"
  name="normalDate">
</sui-date-picker>

<!-- Disabled State -->
<sui-date-picker 
  [(ngModel)]="disabledDate"
  [disabled]="true"
  placeholder="Disabled date"
  name="disabledDate">
</sui-date-picker>

<!-- Readonly State -->
<sui-date-picker 
  [(ngModel)]="readonlyDate"
  [readonly]="true"
  placeholder="Readonly date"
  name="readonlyDate">
</sui-date-picker>
